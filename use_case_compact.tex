\addcontentsline{toc}{subsection}{UC1 Entity creation}
\usecaseheader
  {UC1}  % 1: ID
  {Entity Creation} % 2: Titolo
  {An authorized user wants to add a new entity to the system.} % 3: Goal
  {Scope: Generic CRUD operations. Level: Primary Task.} % 4: Scope
  {The user is logged in. Specific user roles and contextual preconditions depend on the entity being created.} % 5: Preconditions
  {A new entity is created. Any post-creation logic is executed.} % 6: Success
  {The system remains unchanged. The user is prompted with validation or system errors.} % 7: Failed
  {Primary: Authorized User. Secondary: Database.} % 8: Actors
  {The user triggers the creation action for a specific entity from their respective dashboard.} % 9: Trigger
\begin{ucrequirements}
    FR0.1 & Treatment creation \\ \hline
    FR1.1 & Hospital creation \\ \hline
    FR1.4 & Department creation \\ \hline
    FR9.1 & Create insurance policy \\ \hline
    FR29 & Prescription creation \\ \hline
    FR30 & Exemption creation \\ \hline
    FR33 & Examination upload to patient clinical record \\ 
    \hline
    NFR4 & Scalability and capacity \\ \hline   
\end{ucrequirements}
\begin{ucflow}
    1 & The system shows the creation form for the selected entity. \\ \hline
    2 & The user fills in the required fields. \\ \hline
    3 & The user clicks "Save". \\ \hline
    4 & The system performs basic input validation and executes any specific validation. \\ \hline
    5 & The system creates the new record in the database. \\ \hline
    6 & The system executes any post creation logic. \\ \hline
    7 & The system prompts a "Success" message and redirects the user. \\ \hline
\end{ucflow}

\begin{ucextensions}
    3a & \textbf{The user clicks "Cancel":}\newline 
         1. The system aborts the creation process.\newline 
         2. The user is redirected to the previous dashboard. \newline
         3. Use case ends. \\ \hline
    4a & \textbf{Validity check:}\newline 
         1. The system identifies the invalid or missing fields (e.g., duplicate unique constraints).\newline 
         2. The system prompts an error message highlighting the specific fields.\newline 
         3. Go to main flow step 2 to perform a correction. \\ \hline
\end{ucextensions}
\begin{ucsubvariations}
    1a & \textbf{Entity: Treatment}\newline
         \textbf{Specific Preconditions:} The user is logged in as a System Admin.\newline
         \textbf{Specific Validations:} The "name" parameter must be unique within the global catalog.\newline
         \textbf{Post-Creation Logic:} None. \\ \hline
    1b & \textbf{Entity: Hospital structure}\newline
         \textbf{Specific Preconditions:} The user is logged in as a System Admin.\newline
         \textbf{Specific Validations:} The "name" or "address" parameters must be unique.\newline
         \textbf{Post-Creation Logic:} None. \\ \hline
    1c & \textbf{Entity: Department}\newline
         \textbf{Specific Preconditions:} The user is logged in as a System Admin. The specific hospital structure must already exist in the system.\newline
         \textbf{Specific Validations:} The "name" must be unique within the specific hospital structure.\newline
         \textbf{Post-Creation Logic:} The system links the new department to the selected hospital structure. \\ \hline
    1d & \textbf{Entity: Insurance Policy}\newline
         \textbf{Specific Preconditions:} The user is logged in as a System Admin.\newline
         \textbf{Specific Validations:} The unique code must not be duplicated. The discount percentage must be a valid amount.\newline
         \textbf{Post-Creation Logic:} None. \\ \hline
    1e & \textbf{Entity: Prescription}\newline
         \textbf{Specific Preconditions:} The user is logged in as a Doctor.\newline
         \textbf{Specific Validations:} The patient to whom the prescription is assigned must be registered in the system.\newline
         \textbf{Post-Creation Logic:} The System notifies the patient about the creation of the prescription via email. \\ \hline
    1f & \textbf{Entity: Exemption}\newline
         \textbf{Specific Preconditions:} The user is logged in as a Doctor.\newline
         \textbf{Specific Validations:} The patient to whom the exemption is assigned must be registered in the system. The validity period must be logically consistent.\newline
         \textbf{Post-Creation Logic:} The System notifies the patient about the creation of the exemption via email. \\ \hline
    1g & \textbf{Entity: Clinical Record}\newline
         \textbf{Specific Preconditions:} The user is logged in as a Doctor. A medical examination has been performed.\newline
         \textbf{Specific Validations:} The patient to whom the record is related must be registered in the system. The attachment type, if any, must be a compatible format.\newline
         \textbf{Post-Creation Logic:} The System notifies the patient about the upload of the clinical record via email. \\ \hline
\end{ucsubvariations}
